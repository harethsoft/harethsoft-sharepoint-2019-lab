# دالة (Function) لإنشاء المجموعة إذا لم تكن موجودة
function Create-GroupQuick ($gName, $gPath) {
    if (!(Get-ADGroup -Filter "Name -eq '$gName'")) {
        New-ADGroup -Name $gName -GroupScope Global -Path $gPath
    }
}

# 2. إنشاء جميع المجموعات المطلوبة أولاً

Create-GroupQuick "HR_Group" "OU=hr,$baseOU"

# 1. تحديد المسار والجروب الجديد
$baseOU = "OU=SHAREPOINT,DC=harethsoft,DC=com"
$allUsersGroupName = "allusers"

# 2. إنشاء الجروب إذا لم يكن موجوداً
if (!(Get-ADGroup -Filter "Name -eq '$allUsersGroupName'")) {
    New-ADGroup -Name $allUsersGroupName -GroupScope Global -Path $baseOU
    Write-Host "تم إنشاء الجروب $allUsersGroupName بنجاح." -ForegroundColor Green
}

# 3. جلب كافة المستخدمين الذين تبدأ أسماؤهم بـ (fin, hr, it, sd) وإضافتهم للجروب
$users = Get-ADUser -Filter "SamAccountName -like 'fin*' -or SamAccountName -like 'hr*' -or SamAccountName -like 'it*' -or SamAccountName -like 'sd*'"
Add-ADGroupMember -Identity $allUsersGroupName -Members $users

Write-Host "تمت إضافة $($users.Count) مستخدم إلى جروب $allUsersGroupName بنجاح!" -ForegroundColor Cyan
